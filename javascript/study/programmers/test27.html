약수의 개수와 덧셈 27. 두 정수 left와 right가 매개변수로 주어집니다. left부터 right까지의 모든 수들 중에서, 약수의 개수가 짝수인 수는 더하고, 약수의 개수가 홀수인 수는 뺀 수를 return 하도록 solution 함수를 완성해주세요. * 1 ≤ left ≤ right ≤ 1,000 
function solution(left, right) {
    let count = 0;
    for(let i = left; i<=right; i++) {
        let array = [];
        for(let j = 0; j <= i; j++) {
            if(i % j === 0) array.push(j)
        }
        array.length % 2 === 0 ? count += i : count -= i
    }
    return count
}
나의 풀이 for문을 활용하여 나누어 떨어지는 수가 나오면 배열에 저장시킨 후 배열이 짝수라면 count 값에 +, 그렇지 않다면 -로 넣어주었습니다. 
function solution(left, right) {
    var answer = 0;
    for (let i = left; i <= right; i++) {
        if (Number.isInteger(Math.sqrt(i))) {
            answer -= i;
        } else {
            answer += i;
        }
    }
    return answer;
}
