핸드폰 번호 가리기 07. 프로그래머스 모바일은 개인정보 보호를 위해 고지서를 보낼 때 고객들의 전화번호의 일부를 가립니다. 전화번호가 문자열 phone_number로 주어졌을 때, 전화번호의 뒷 4자리를 제외한 나머지 숫자를 전부 *으로 가린 문자열을 리턴하는 함수, solution을 완성해주세요. * N의 범위 : phone_number는 길이 4 이상, 20이하인 문자열입니다. 
function solution(phone_number) {
    const pnum = phone_number.length - 4;
    return "*".repeat(pnum) + phone_number.slice(-4);
}
나의 풀이 핸드폰 번호에서 뒤에 4자리를 뺀 길이에는 * 표시를 반복시켜주고 나머지 뒤에서부터 4자리는 번호를 출력할 수 있도록 하였습니다. 
function hide_numbers(s) {
    return s.replace(/\d(?=\d{4})/g, "*");
    }
다른 풀이 정규식을 활용하여 만들 수도 있습니다. 
function hide_numbers(s){
    var result = ""
    for(var i=0;i < s.length;i++){    
      result += i < s.length -4 ? "*" : s.charAt(i);
    }  
    return result;
}
다른 풀이 for문과 삼항연산식을 활용한 풀이입니다. 
const solution = n => [...n].fill("*",0,n.length-4).join("")
